-- bases table
CREATE TABLE bases (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  slug TEXT UNIQUE NOT NULL,
  townhall INTEGER CHECK (townhall BETWEEN 3 AND 18),
  base_type TEXT CHECK (base_type IN (
    'War','Farming','Hybrid','Trophy',
    'Anti 3 Star','Anti Air','Legend League','Progress'
  )),
  image_url TEXT NOT NULL,
  layout_link TEXT NOT NULL,
  description TEXT NOT NULL,
  views INTEGER DEFAULT 0,
  copies INTEGER DEFAULT 0,
  report_count INTEGER DEFAULT 0,
  is_active BOOLEAN DEFAULT true,
  key_features TEXT[] DEFAULT '{}',
  best_against TEXT[] DEFAULT '{}',
  win_rate INTEGER DEFAULT 80,
  difficulty TEXT DEFAULT 'Medium',
  rating_avg DECIMAL DEFAULT 0,
  rating_count INTEGER DEFAULT 0,
  health_score INTEGER GENERATED ALWAYS AS (
    GREATEST(0, win_rate - (report_count * 5))
  ) STORED,
  approved BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- RLS Policies
ALTER TABLE bases ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Public read approved bases" ON bases
  FOR SELECT USING (approved = true AND is_active = true);
CREATE POLICY "Admin full access" ON bases
  FOR ALL USING (
    auth.jwt() ->> 'email' = current_setting('app.admin_email')
  );

-- blog_posts table
CREATE TABLE blog_posts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  slug TEXT UNIQUE NOT NULL,
  content TEXT NOT NULL,
  excerpt TEXT NOT NULL,
  featured_image TEXT NOT NULL,
  views INTEGER DEFAULT 0,
  meta_title TEXT,
  meta_description TEXT,
  published BOOLEAN DEFAULT true,
  published_at TIMESTAMP DEFAULT NOW(),
  created_at TIMESTAMP DEFAULT NOW()
);

-- submissions table
CREATE TABLE submissions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id),
  townhall INTEGER,
  base_type TEXT,
  layout_link TEXT NOT NULL,
  description TEXT,
  image_url TEXT,
  status TEXT DEFAULT 'pending' 
    CHECK (status IN ('pending','approved','rejected')),
  admin_notes TEXT,
  submitted_at TIMESTAMP DEFAULT NOW()
);

-- comments table
CREATE TABLE comments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  base_id UUID REFERENCES bases(id) ON DELETE CASCADE,
  user_id UUID REFERENCES auth.users(id),
  user_name TEXT NOT NULL,
  content TEXT NOT NULL,
  rating INTEGER CHECK (rating BETWEEN 1 AND 5),
  approved BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW()
);

-- reports table (NEW - unique feature)
CREATE TABLE reports (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  base_id UUID REFERENCES bases(id) ON DELETE CASCADE,
  reason TEXT CHECK (reason IN (
    'broken_link','outdated','wrong_th','spam','other'
  )),
  reporter_ip TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- views_analytics table
CREATE TABLE views_analytics (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  base_id UUID REFERENCES bases(id),
  viewed_at DATE DEFAULT CURRENT_DATE,
  count INTEGER DEFAULT 1,
  UNIQUE(base_id, viewed_at)
);

-- saved_bases table
CREATE TABLE saved_bases (
  user_id UUID REFERENCES auth.users(id),
  base_id UUID REFERENCES bases(id) ON DELETE CASCADE,
  saved_at TIMESTAMP DEFAULT NOW(),
  PRIMARY KEY (user_id, base_id)
);
